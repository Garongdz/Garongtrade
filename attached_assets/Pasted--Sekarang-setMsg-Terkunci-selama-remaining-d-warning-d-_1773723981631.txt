import bcrypt, os
from datetime import datetime, timedelta
from jose import jwt, JWTError

JWT_SECRET = os.environ.get("JWT_SECRET", "fallback")
APP_PIN = os.environ.get("APP_PIN", "000000")
COOKIE_NAME = "gs_session"

PIN_HASH = bcrypt.hashpw(APP_PIN.encode(), bcrypt.gensalt())
login_attempts = {}

def verify_pin(pin: str) -> bool:
    return bcrypt.checkpw(pin.encode(), PIN_HASH)

def create_session_token() -> str:
    expire = datetime.utcnow() + timedelta(days=7)
    return jwt.encode(
        {"exp": expire, "type": "session"},
        JWT_SECRET, algorithm="HS256"
    )

def verify_session_token(token: str) -> bool:
    try:
        jwt.decode(token, JWT_SECRET, algorithms=["HS256"])
        return True
    except JWTError:
        return False

def is_rate_limited(ip: str) -> dict:
    now = datetime.utcnow()
    record = login_attempts.get(ip, 
             {"count": 0, "locked_until": None})
    if record["locked_until"] and now < record["locked_until"]:
        remaining = int(
            (record["locked_until"] - now).total_seconds()
        )
        return {"limited": True, "remaining": remaining}
    if record["locked_until"] and now >= record["locked_until"]:
        login_attempts[ip] = {"count": 0, "locked_until": None}
    return {"limited": False, "remaining": 0}

def record_failed_attempt(ip: str) -> dict:
    record = login_attempts.get(ip, 
             {"count": 0, "locked_until": None})
    record["count"] += 1
    if record["count"] >= 3:
        record["locked_until"] = (datetime.utcnow() 
                                  + timedelta(seconds=30))
        record["count"] = 0
        login_attempts[ip] = record
        return {"locked": True, "attempts_left": 0}
    login_attempts[ip] = record
    return {"locked": False, 
            "attempts_left": 3 - record["count"]}

from fastapi import APIRouter, Request, Response

from auth import (verify_pin, create_session_token, 
                  is_rate_limited, record_failed_attempt,
                  COOKIE_NAME)

async def login_page():

        return f.read()

@router.post("/api/auth/login")
async def handle_login(request: Request):
    body = await request.json()
    pin = body.get("pin", "")
    ip = request.client.host

    rate = is_rate_limited(ip)
    if rate["limited"]:
        return JSONResponse(
            {"success": False, "locked": True, 
             "remaining": rate["remaining"]},
            status_code=429
        )

    if verify_pin(pin):
        from auth import login_attempts
        login_attempts.pop(ip, None)
        token = create_session_token()
        response = JSONResponse({"success": True})
        response.set_cookie(
            key=COOKIE_NAME,
            value=token,
            httponly=True,
            max_age=7 * 24 * 60 * 60,
            samesite="lax",
            secure=False
        )
        return response
    else:
        result = record_failed_attempt(ip)
        return JSONResponse(
            {"success": False,
             "locked": result["locked"],
             "attempts_left": result["attempts_left"]},
            status_code=401
        )

@router.post("/api/auth/logout")
async def handle_logout():
    response = JSONResponse({"success": True})
    response.delete_cookie(COOKIE_NAME)
    return response

## FILE: middleware/auth.py
from fastapi import Request
from fastapi.responses import RedirectResponse, JSONResponse
from starlette.middleware.base import BaseHTTPMiddleware
from auth import verify_session_token, COOKIE_NAME

EXEMPT_PATHS = [
    "/login",
    "/api/auth/login",
    "/api/auth/logout",
    "/api/health",
    "/favicon.ico",
]

class AuthMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        path = request.url.path
        if any(path.startswith(p) for p in EXEMPT_PATHS):
            return await call_next(request)
        token = request.cookies.get(COOKIE_NAME)
        if not token or not verify_session_token(token):
            if path.startswith("/api/"):
                return JSONResponse(
                    {"error": "Unauthorized"}, 
                    status_code=401
                )
            return RedirectResponse(url="/login")
        return await call_next(request)

(function() {
  var pin = '';
  var locked = false;
  var submitting = false;

  function updateDots(state) {
    for (var i = 0; i < 6; i++) {
      var dot = document.getElementById('d' + i);
      dot.className = 'dot';
      if (state === 'error' && i < pin.length) {
        dot.classList.add('error');
      } else if (state === 'success') {
        dot.classList.add('success');
      } else if (i < pin.length) {
        dot.classList.add('filled');
        dot.classList.add('pop');
      }
    }
  }

  function setMsg(text, type) {
    var el = document.getElementById('msg');
    el.textContent = text;
    el.className = 'msg ' + (type || '');
  }

  function shake() {
    var el = document.getElementById('dots');
    el.classList.remove('shake');
    void el.offsetWidth;
    el.classList.add('shake');
  }

  function pressNum(num) {
    if (locked || submitting || pin.length >= 6) return;
    pin += num;
    updateDots();
    if (pin.length === 6) submitPin();
  }

  function backspace() {
    if (locked || submitting) return;
    pin = pin.slice(0, -1);
    updateDots();
    setMsg('');
  }

  function submitPin() {
    submitting = true;
    fetch('/api/auth/login', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({pin: pin})
    })
    .then(function(res) { return res.json(); })
    .then(function(data) {
      if (data.success) {
        updateDots('success');
        setMsg('Selamat datang!', 'success');
        setTimeout(function() {
          window.location.href = '/';
        }, 800);
      } else if (data.locked) {
        locked = true;
        updateDots('error');
        shake();
        var remaining = data.remaining;
        var timer = setInterval(function() {
          setMsg('Terkunci selama ' + remaining 
                 + ' detik', 'warning');
          remaining--;
          if (remaining < 0) {
            clearInterval(timer);
            locked = false;
            pin = '';
            updateDots();
            setMsg('');
          }
        }, 1000);
      } else {
        updateDots('error');
        shake();
        var left = data.attempts_left;
        if (left === 1) {
          setMsg('PIN salah — 1 percobaan tersisa', 'error');
        } else {
          setMsg('PIN salah, coba lagi', 'error');
        }
        setTimeout(function() {
          pin = '';
          updateDots();
        }, 600);
      }
      submitting = false;
    })
    .catch(function() {
      setMsg('Koneksi gagal, coba lagi', 'error');
      pin = '';
      updateDots();
      submitting = false;
    });
  }

  document.querySelectorAll('.num-btn').forEach(function(btn) {
    btn.addEventListener('click', function() {
      var val = this.getAttribute('data-val');
      if (!val) return;
      if (val === 'back') backspace();
      else pressNum(val);
    });
  });
})();
